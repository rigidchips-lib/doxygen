823 //
824 // IVMRDeinterlaceControl
825 //
826 // New interfaced introduced into the WindowsXP SP1 release of the VMR.
827 // This interface allows applications to control the DX-VA deinterlacing
828 // support provided by the VMR.
829 //
830 // The VMR needs to be set into "mixing" mode for this interface to work.
831 //
832 // SetDeinterlaceMode is only effective for new connections made to the
833 // VMR. It should be noted that the graphics device driver may refuse
834 // to use the specified deinterlace mode, in which case 3 fallback
835 // policies are offered by the VMR, these being:
836 //
837 // 1. Fallback to the next best mode offered by the driver.
838 // 2. Fallback to the BOB deinterlace mode.
839 // 3. Fallback to the WEAVE deinterlace mode (ie. turn deinterlacing off).
840 //

857  // isn't recommended

859 
860 
861  // the algorithm creates the missing lines by vertically stretching each
862  // video field by a factor of two, for example by averaging two lines or
863  // using a [-1, 9, 9, -1]/16 filter across four lines.
864  // Slight vertical adjustments are made to ensure that the resulting image
865  // does not "bob" up and down.
866  DeinterlaceTech9_BOBVerticalStretch = 0x0002,
867 
868  // the pixels in the missing line are recreated by a median filtering operation
869  DeinterlaceTech9_MedianFiltering = 0x0004,
870 
871  // the pixels in the missing line are recreated by an edge filter.
872  // In this process, spatial directional filters are applied to determine
873  // the orientation of edges in the picture content, and missing
874  // pixels are created by filtering along (rather than across) the
875  // detected edges.
876  DeinterlaceTech9_EdgeFiltering = 0x0010,
877 
878  // the pixels in the missing line are recreated by switching on a field by
879  // field basis between using either spatial or temporal interpolation
880  // depending on the amount of motion.
881  DeinterlaceTech9_FieldAdaptive = 0x0020,
882 
883  // the pixels in the missing line are recreated by switching on a pixel by pixel
884  // basis between using either spatial or temporal interpolation depending on
885  // the amount of motion..

888  // Motion Vector Steering identifies objects within a sequence of video
